SUM1	EQU 033h
SUM2	EQU 066h

	MOV SUM1-3, #0FEh
	MOV SUM1-2, #0EDh
	MOV SUM1-1, #0BEh
	MOV SUM1, #0EFh
	
	MOV SUM2-3, #0ABh
	MOV SUM2-2, #0BAh
	MOV SUM2-1, #0DEh
	MOV SUM2, #0ADh

	MOV R0, #SUM1
	MOV R1, #SUM2
	CALL ADD32

FINAL:	JMP FINAL

ADD32:
	MOV R2, #04h
	CLR C
LOOP:	MOV A, @R0
	ADDC A, @R1
	MOV @R0, A
	DEC R0
	DEc R1
	DJNZ R2, LOOP

	RET
